def convert_bounds(bbox, invert_y=False):
    """
    Helper method for changing bounding box representation to leaflet notation

    ``(lon1, lat1, lon2, lat2) -> ((lat1, lon1), (lat2, lon2))``

    x1, y1, x2, y2 = bbox
    if invert_y:

    return ((y1, x1), (y2, x2))
